import streamlit as st
import pandas as pd
import json
from pathlib import Path
from PIL import Image
import uuid
from datetime import datetime

# Local storage
DATA_FILE = Path("sports_collection.json") PHOTO_DIR = Path("sports_card_photos")
PHOTO_DIR.mkdir(exist_ok=True)

def load_collection():
    if DATA_FILE.exists():
        with open(DATA_FILE, "r", encoding="utf-8") as f:
            return json.load(f)
    return []

def save_collection(collection):
    with open(DATA_FILE, "w", encoding="utf-8") as f:
        json.dump(collection, f, indent=2)

def save_photo(uploaded_file, card_id: str):
    if not uploaded_file:
        return None
    ext = uploaded_file.name.split(".")[-1].lower()
    photo_path = PHOTO_DIR / f"{card_id}.{ext}"
    image = Image.open(uploaded_file).convert("RGB")
    image.save(photo_path)
    return str(photo_path)

st.set_page_config(page_title="CardVault Sports", layout="wide")
st.title("🏈 CardVault Sports")
st.caption("Personal Sports Card Collection • Beckett Checklist Import • Set Completion • Custom Categories")

if "collection" not in st.session_state:
    st.session_state.collection = load_collection()

collection = st.session_state.collection

tab_add, tab_import, tab_collection = st.tabs(["➕ Add Single Card", "📥 Import Beckett Checklist", "📦 My Collection"])

with tab_add:
    st.subheader("Add a Single Sports Card")
    with st.form("add_card_form", clear_on_submit=True):
        col1, col2 = st.columns(2)
        with col1:
            player = st.text_input("Player Name *", placeholder="Aaron Judge")
            sport = st.selectbox("Sport *", ["Baseball", "Football", "Basketball", "Hockey", "Other"])
            set_name = st.text_input("Set / Year *", placeholder="2024 Topps Series 1")
        with col2:
            card_number = st.text_input("Card Number", placeholder="1")
            team = st.text_input("Team", placeholder="New York Yankees")
            rarity = st.text_input("Rarity / Parallel", placeholder="Rookie Refractor")
        
        category = st.text_input("Custom Category (optional)", placeholder="Rookie Cards, Yankees PC")
        notes = st.text_area("Notes", placeholder="Condition, grade...")
        uploaded_photo = st.file_uploader("Upload photo (optional)", type=["jpg", "jpeg", "png"])
        
        if st.form_submit_button("Add to Collection", type="primary"):
            if not player or not set_name or not sport:
                st.error("Player, Sport, and Set required!")
            else:
                card_id = str(uuid.uuid4())
                photo_path = save_photo(uploaded_photo, card_id)
                new_card = {
                    "id": card_id,
                    "player": player.strip(),
                    "sport": sport,
                    "set": set_name.strip(),
                    "number": card_number.strip(),
                    "team": team.strip(),
                    "rarity": rarity.strip(),
                    "category": category.strip() if category.strip() else None,
                    "notes": notes.strip(),
                    "custom_photo": photo_path,
                    "added_date": datetime.now().isoformat()
                }
                collection.append(new_card)
                st.session_state.collection = collection
                save_collection(collection)
                st.success(f"✅ Added {player}")
                st.rerun()

with tab_import:
    st.subheader("Import from Beckett Checklist (XLSX)")
    st.info("Download the checklist XLSX from Beckett.com, then upload it here.")
    uploaded_xlsx = st.file_uploader("Upload Beckett Checklist XLSX", type=["xlsx", "xls"])
    
    if uploaded_xlsx:
        try:
            df = pd.read_excel(uploaded_xlsx)
            st.success(f"Loaded {len(df)} rows")
            st.dataframe(df.head(10), use_container_width=True)
            
            set_name_guess = st.text_input("Set Name for these cards *", placeholder="2025 Topps Series 1")
            sport_guess = st.selectbox("Sport for these cards *", ["Baseball", "Football", "Basketball", "Hockey", "Other"])
            
            if st.button("Import as Unowned Cards", type="primary"):
                if not set_name_guess or not sport_guess:
                    st.error("Set Name and Sport required!")
                else:
                    imported_count = 0
                    for _, row in df.iterrows():
                        player_val = str(row.get("Player", row.get("Name", "")))
                        num_val = str(row.get("Card #", row.get("Number", "")))
                        team_val = str(row.get("Team", ""))
                        if player_val.strip():
                            card_id = str(uuid.uuid4())
                            new_card = {
                                "id": card_id,
                                "player": player_val.strip(),
                                "sport": sport_guess,
                                "set": set_name_guess.strip(),
                                "number": num_val.strip(),
                                "team": team_val.strip(),
                                "rarity": "",
                                "category": None,
                                "notes": "Imported from Beckett",
                                "custom_photo": None,
                                "added_date": datetime.now().isoformat()
                            }
                            collection.append(new_card)
                            imported_count += 1
                    st.session_state.collection = collection
                    save_collection(collection)
                    st.success(f"✅ Imported {imported_count} cards!")
                    st.rerun()
        except Exception as e:
            st.error(f"Error: {e}")

with tab_collection:
    st.subheader(f"My Sports Collection ({len(collection)} cards)")
    if not collection:
        st.info("No cards yet — add or import some!")
    else:
        df_data = []
        for c in collection:
            df_data.append({
                "Player": c["player"],
                "Sport": c["sport"],
                "Set": c["set"],
                "Card #": c.get("number", ""),
                "Team": c.get("team", ""),
                "Rarity": c.get("rarity", ""),
                "Category": c.get("category") or "—",
                "Has Photo": "Yes" if c.get("custom_photo") else "No",
                "Notes": (c.get("notes") or "")[:60] + ("..." if len(c.get("notes", "")) > 60 else "")
            })
        df = pd.DataFrame(df_data)
        
        st.write("### Set Completion Overview")
        set_groups = df.groupby("Set").size().reset_index(name="Cards Owned")
        set_groups = set_groups.sort_values("Cards Owned", ascending=False)
        cols = st.columns(3)
        for idx, row in set_groups.iterrows():
            with cols[idx % 3]:
                st.metric(row["Set"], f"{row['Cards Owned']} cards")
        
        st.divider()
        
        # Filters
        col1, col2, col3 = st.columns([3, 2, 2])
        with col1:
            search = st.text_input("🔍 Search player, set, or team")
        with col2:
            selected_sports = st.multiselect("Sport", sorted(df["Sport"].unique()), default=[])
        with col3:
            selected_categories = st.multiselect("Category", sorted([cat for cat in df["Category"].unique() if cat != "—"]), default=[])
        
        filtered_df = df.copy()
        if search:
            mask = filtered_df["Player"].str.contains(search, case=False, na=False) | filtered_df["Set"].str.contains(search, case=False, na=False)
            filtered_df = filtered_df[mask]
        if selected_sports:
            filtered_df = filtered_df[filtered_df["Sport"].isin(selected_sports)]
        if selected_categories:
            filtered_df = filtered_df[filtered_df["Category"].isin(selected_categories)]
        
        st.dataframe(filtered_df, use_container_width=True, hide_index=True)
        
        st.divider()
        st.write("### Manage Cards (click to expand)")
        for card in sorted(collection, key=lambda x: (x.get("set", ""), x.get("player", ""))):
            with st.expander(f"{card['player']} — {card.get('set', '')}"):
                if card.get("custom_photo"):
                    st.image(card["custom_photo"], width=280)
                st.write(f"Team: {card.get('team', '—')} | # {card.get('number', '—')} | Rarity: {card.get('rarity', '—')}")
                st.write(f"Category: {card.get('category') or 'None'}")
                if card.get("notes"):
                    st.write(f"Notes: {card['notes']}")
                if st.button("🗑️ Remove", key=f"del_{card['id']}"):
                    st.session_state.collection = [c for c in st.session_state.collection if c["id"] != card["id"]]
                    save_collection(st.session_state.collection)
                    st.rerun()

st.sidebar.success("CardVault Sports Ready")
